CREATE TABLE stocks (
  product_id SERIAL PRIMARY KEY,         -- 商品ID（自動採番）
  product_name TEXT NOT NULL,            -- 商品名
  quantity INTEGER NOT NULL DEFAULT 0,   -- 商品個数
  deleted BOOLEAN NOT NULL DEFAULT FALSE -- デリートフラグ（削除済みかどうか）
);
